Problem 1

For this problem, I generated a normal distributed background with a sigma value 3 and a uniform distributed signal with strength 15.

In [47]:
%matplotlib inline
import matplotlib
import matplotlib.pyplot as plt
import numpy as np
import matplotlib.pyplot as plt
import scipy as sp
from scipy import stats
plt.rcParams["figure.figsize"] = (15,10)

d = stats.norm.rvs(loc=0,scale=3, size = 1000000)
sig = stats.uniform.rvs(loc=0,scale=15, size = 1000000)

data = d + sig

signaledges = np.linspace(0,15,40)
dataedges = np.linspace(0,20,40)

Psd, temp, temp2= np.histogram2d(data,sig, bins=[dataedges,signaledges], density=True)

datacenters = (dataedges[:-1] + dataedges[1:]) / 2
signalcenters = (signaledges[:-1] + signaledges[1:]) / 2
plt.pcolormesh(datacenters,signalcenters,Psd.T)
plt.ylabel('True signal, $P(s|d)$', fontsize = 24)
plt.xlabel('Observed data, $P(d|s)$', fontsize = 24)
